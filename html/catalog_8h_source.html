191  template <class KeyType, class ValueType, class KeyComparator>
192  IndexInfo *CreateIndex(Transaction *txn, const std::string &index_name, const std::string &table_name,
193  const Schema &schema, const Schema &key_schema, const std::vector<uint32_t> &key_attrs,
194  std::size_t keysize, HashFunction<KeyType> hash_function) {
195  // Reject the creation request for nonexistent table
196  if (table_names_.find(table_name) == table_names_.end()) {
197  return NULL_INDEX_INFO;
198  }
199 
200  // If the table exists, an entry for the table should already be present in index_names_
201  BUSTUB_ASSERT((index_names_.find(table_name) != index_names_.end()), "Broken Invariant");
202 
203  // Determine if the requested index already exists for this table
204  auto &table_indexes = index_names_.find(table_name)->second;
205  if (table_indexes.find(index_name) != table_indexes.end()) {
206  // The requested index already exists for this table
207  return NULL_INDEX_INFO;
208  }
209 
210  // Construct index metdata
211  auto meta = std::make_unique<IndexMetadata>(index_name, table_name, &schema, key_attrs);
212 
213  // Construct the index, take ownership of metadata
214  // TODO(Kyle): We should update the API for CreateIndex
215  // to allow specification of the index type itself, not
216  // just the key, value, and comparator types
217  auto index = std::make_unique<ExtendibleHashTableIndex<KeyType, ValueType, KeyComparator>>(std::move(meta), bpm_,
218  hash_function);
219 
220  // Populate the index with all tuples in table heap
221  auto *table_meta = GetTable(table_name);
222  auto *heap = table_meta->table_.get();
223  for (auto tuple = heap->Begin(txn); tuple != heap->End(); ++tuple) {
224  index->InsertEntry(tuple->KeyFromTuple(schema, key_schema, key_attrs), tuple->GetRid(), txn);
225  }
226 
227  // Get the next OID for the new index
228  const auto index_oid = next_index_oid_.fetch_add(1);
229 
230  // Construct index information; IndexInfo takes ownership of the Index itself
231  auto index_info =
232  std::make_unique<IndexInfo>(key_schema, index_name, std::move(index), index_oid, table_name, keysize);
233  auto *tmp = index_info.get();
234 
235  // Update internal tracking
236  indexes_.emplace(index_oid, std::move(index_info));
237  table_indexes.emplace(index_name, index_oid);
238 
239  return tmp;
240  }
