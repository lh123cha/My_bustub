76  void CombineAggregateValues(AggregateValue *result, const AggregateValue &input) {
77  for (uint32_t i = 0; i < agg_exprs_.size(); i++) {
78  switch (agg_types_[i]) {
79  case AggregationType::CountAggregate:
80  // Count increases by one.
81  result->aggregates_[i] = result->aggregates_[i].Add(ValueFactory::GetIntegerValue(1));
82  break;
83  case AggregationType::SumAggregate:
84  // Sum increases by addition.
85  result->aggregates_[i] = result->aggregates_[i].Add(input.aggregates_[i]);
86  break;
87  case AggregationType::MinAggregate:
88  // Min is just the min.
89  result->aggregates_[i] = result->aggregates_[i].Min(input.aggregates_[i]);
90  break;
91  case AggregationType::MaxAggregate:
92  // Max is just the max.
93  result->aggregates_[i] = result->aggregates_[i].Max(input.aggregates_[i]);
94  break;
95  }
